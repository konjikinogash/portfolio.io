Water Supply Management System For Crops

Challenges

  • Irregular rainfall patterns affecting water availability.
  • Rapid urbanization and increasing population putting pressure on water resources.
  • Limited water resources available for both urban and agricultural needs.
  • Need for efficient water distribution to reduce wastage and improve access.
  • Importance of accurate and timely data for decision-making.

Solutions

  • Data acquisition and transmission infrastructure to collect and send data from sensors.
  • Data storage and processing through databases and cloud computing.
  • Algorithms like time series analysis and predictive modeling to forecast water levels and predict issues.
  • A user-friendly dashboard for monitoring water levels and triggering alerts.
  • Integration with existing water supply systems for better management and control.

Objectives

  • Identify areas with high water consumption and optimize water distribution.
  • Plan water supply for agricultural crops based on resource availability.
  • Implement data structures for efficient monitoring and planning.

#include <iostream>
#include <vector>
#include <climits>
#include <algorithm>
using namespace std;

// Function to identify areas with high water consumption
void identifyHighConsumption(vector<pair<int, string>> &areas) {
    sort(areas.begin(), areas.end(), greater<>());
    cout << "Areas with high water consumption:\n";
    for (auto &area : areas) {
        cout << "Area: " << area.second << ", Consumption: " << area.first << " liters\n";
    }
}

// Function to find the shortest path in a graph
void findShortestPath(int nodes, vector<vector<int>> &graph, int source) {
    vector<int> distance(nodes, INT_MAX);
    distance[source] = 0;

    for (int i = 0; i < nodes - 1; i++) {
        for (int u = 0; u < nodes; u++) {
            for (int v = 0; v < nodes; v++) {
                if (graph[u][v] != 0 && distance[u] != INT_MAX &&
                    distance[u] + graph[u][v] < distance[v]) {
                    distance[v] = distance[u] + graph[u][v];
                }
            }
        }
    }

    cout << "\nShortest distances from source " << source << ":\n";
    for (int i = 0; i < nodes; i++) {
        cout << "Node " << i << ": " << (distance[i] == INT_MAX ? -1 : distance[i]) << "\n";
    }
}

int main() {
    vector<pair<int, string>> areas = {
        {3000, "Area A"},
        {4500, "Area B"},
        {2000, "Area C"},
        {5000, "Area D"}
    };
    identifyHighConsumption(areas);

    vector<vector<int>> graph = {
        {0, 10, 0, 5},
        {0, 0, 1, 2},
        {0, 0, 0, 0},
        {0, 3, 9, 0}
    };
    findShortestPath(4, graph, 0);

    return 0;
}
